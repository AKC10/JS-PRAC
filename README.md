# JS-PRAC
//temperature in kelvin
const kelvin = 0;
//conversion K --> C
let celsius = kelvin - 273; 
// fahrenheit conversion
let Fahrenheit = celsius * (9/5) + 32;
//decimal variable
Fahrenheit = Math.floor(Fahrenheit);
console.log (`The temperature is ${Fahrenheit} degrees Fahrenheit.`);
